# SCL Library CMAKE file

CMAKE_MINIMUM_REQUIRED(VERSION 3.14 FATAL_ERROR)

PROJECT(SCL)

SET(CMAKE_CXX_STANDARD 20)
SET(CMAKE_CXX_STANDARD_REQUIRED True)

SET(LIBRARY_OUTPUT_PATH ${CMAKE_BINARY_DIR}/SCL)

INCLUDE_DIRECTORIES(${CMAKE_CURRENT_LIST_DIR})
INCLUDE_DIRECTORIES(${GCL_SOURCE_DIR})
INCLUDE_DIRECTORIES(${MCL_SOURCE_DIR})

ADD_LIBRARY(SCL ${PROJECT_SOURCE_DIR}/source/array1DP.cpp
                ${PROJECT_SOURCE_DIR}/source/array2DP.cpp
                ${PROJECT_SOURCE_DIR}/source/common.cpp
                ${PROJECT_SOURCE_DIR}/source/config.cpp
                ${PROJECT_SOURCE_DIR}/source/valarray2DP.cpp)
